let estudiante1Cursos = ['Matemáticas', 'Inglés', 'Programación']; let estudiante2Cursos = ['Geografía', 'Español', 'Programación'];
function materias(arr, arr2){
let resultado;
for(materia of arr){
for(mate of arr2){

resultado = mate;
}
}
}

}
